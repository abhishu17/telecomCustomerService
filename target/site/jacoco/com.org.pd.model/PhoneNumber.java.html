package com.org.pd.model;

import java.util.List;

public class PhoneNumber {
	private int customerId;
	private String phoneNumber;
	private boolean activationstatus;
	
	public PhoneNumber() {

	}

	public PhoneNumber(int customerId, String phoneNumber, boolean status) {
		super();
		this.customerId = customerId;
		this.phoneNumber = phoneNumber;
		this.activationstatus = status;
	}
	
	

	public int getCustomerId() {
		return customerId;
	}

	public void setCustomerId(int customerId) {
		this.customerId = customerId;
	}

	public String getPhoneNumber() {
		return phoneNumber;
	}

	public void setPhoneNumber(String phoneNumber) {
		this.phoneNumber = phoneNumber;
	}

	
	public boolean isActivationstatus() {
		return activationstatus;
	}

	public void setActivationstatus(boolean activationstatus) {
		this.activationstatus = activationstatus;
	}

	@Override
	public String toString() {
		return "PhoneNumber [customerId=" + customerId + ", phoneNumber=" + phoneNumber + ", activationstatus=" + activationstatus + "]";
	}

	@Override
	public int hashCode() {
		final int prime = 31;
		int result = 1;
		result = prime * result + ((customerId == 0) ? 0 : phoneNumber.hashCode());
		return result;
	}

	@Override
	public boolean equals(Object obj) {
		if (this == obj)
			return true;
		if (obj == null)
			return false;
		if (getClass() != obj.getClass())
			return false;
		PhoneNumber other = (PhoneNumber) obj;
		if (customerId == 0) {
			if (other.customerId != 0)
				return false;
		} else if (customerId != customerId)
			return false;
		return true;
	}

}
